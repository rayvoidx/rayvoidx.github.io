# 색인 가치가 없는 경로 — 크롤링 예산 절약.
# privacy.html 은 문서 자체에 <meta name="robots" content="noindex"> 가 이미 있다.
Disallow: /*/privacy.html
Disallow: /*/404.html

# 저장소 내부 파일 — worker/index.js 의 BLOCKED 정규식도 같은 것을 막는다(이중 방어).
Disallow: /*/SERVICE.yaml
Disallow: /*/WIKI.md

# 크롤러가 만질 수 있는 상태 파라미터는 결과 공유용이라 색인할 필요가 없다.
Disallow: /*?*lang=
